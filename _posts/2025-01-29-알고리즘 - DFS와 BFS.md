dy = [0, 0, -1, 1]

def bfs(x, y):
    queue = deque()
    queue.append((x, y))
    visited[x][y] = True

        x, y = queue.popleft()
        for i in range(4):
            nx, ny = x + dx[i], y + dy[i]
            if 0 <= nx < n and 0 <= ny < m:
                if maze[nx][ny] == 1 and not visited[nx][ny]:
                    maze[nx][ny] = maze[x][y] + 1
                    visited[nx][ny] = True
                    queue.append((nx, ny))

## ğŸ§ª ì˜ˆì œ 3: [ë°±ì¤€ 7569 - í† ë§ˆí†  (3ì°¨ì› BFS)](https://www.acmicpc.net/problem/7569)

queue = deque()

dy = [0, 0, -1, 1, 0, 0]
dz = [0, 0, 0, 0, -1, 1]

    for x in range(n):
        for y in range(m):
            if box[z][x][y] == 1:
                queue.append((z, x, y))

while queue:
    z, x, y = queue.popleft()
    for i in range(6):
        nz, nx, ny = z + dz[i], x + dx[i], y + dy[i]
        if 0 <= nz < h and 0 <= nx < n and 0 <= ny < m:
            if box[nz][nx][ny] == 0:
                box[nz][nx][ny] = box[z][x][y] + 1
                queue.append((nz, nx, ny))

res = 0
for z in box:
    for row in z:
        for val in row:
            if val == 0:
                print(-1)
                exit(0)
        res = max(res, max(row))
