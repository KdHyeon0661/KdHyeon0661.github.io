import torch

def sample_curve(r_scalar_func, t_vals):
    """
    r_scalar_func: t (scalar tensor, requires_grad=True) -> (3,) position
    t_vals: (N,) float tensor
    returns: dict with r, v, T, a, kappa, (optional) tau, N, B
    """
    r_list, v_list, a_list = [], [], []
    for tv in t_vals:
        t = tv.clone().detach().requires_grad_(True)

        r = r_scalar_func(t)        # (3,)
        # first derivative
        dr = torch.autograd.grad(r, t, grad_outputs=torch.ones_like(r), create_graph=True)[0]  # scalar
        # "dr" is sum_j dr_j/dt; we need component-wise. Trick: sum of components with unit vectors.
        # Better: compute dr by differentiating each component individually:
        # Implement component-wise derivative:
        comps = []
        for i in range(r.numel()):
            gi = torch.zeros_like(r); gi[i] = 1.0
            dri = torch.autograd.grad(r, t, grad_outputs=gi, retain_graph=True, create_graph=True)[0]
            comps.append(dri)
        dr_vec = torch.stack(comps)              # (3,)
        # second derivative
        comps2 = []
        for i in range(dr_vec.numel()):
            dri = dr_vec[i]
            d2ri = torch.autograd.grad(dri, t, retain_graph=True, create_graph=False)[0]
            comps2.append(d2ri)
        d2r_vec = torch.stack(comps2)            # (3,)

        r_list.append(r.detach())
        v_list.append(dr_vec.detach())
        a_list.append(d2r_vec.detach())

    r = torch.stack(r_list)      # (N,3)
    v = torch.stack(v_list)      # (N,3)
    a = torch.stack(a_list)      # (N,3)

    speed = torch.linalg.vector_norm(v, dim=-1)            # (N,)
    T = v / speed.unsqueeze(-1).clamp_min(1e-12)           # (N,3)
    # curvature (3D): ||v x a|| / ||v||^3
    cross = torch.cross(v, a, dim=-1)
    kappa = torch.linalg.vector_norm(cross, dim=-1) / (speed**3).clamp_min(1e-12)

    # N, B (where kappa>0)
    N = torch.zeros_like(T)
    pos = (kappa > 1e-12)
    Tdiff = torch.zeros_like(T)
    # numerical derivative of T wrt index (central diff)
    if T.shape[0] >= 3:
        Tdiff[1:-1] = T[2:] - T[:-2]
        N[pos] = normalize(Tdiff[pos])  # direction only (scale not equal to dT/ds)
    # binormal
    B = torch.cross(T, N, dim=-1)

    return {
        "r": r, "v": v, "a": a, "speed": speed,
        "T": T, "N": N, "B": B, "kappa": kappa
    }
