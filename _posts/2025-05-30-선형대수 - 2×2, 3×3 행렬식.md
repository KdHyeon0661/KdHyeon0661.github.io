import numpy as np

def orient2d(A, B, C, eps=1e-12):
    x1,y1 = A; x2,y2 = B; x3,y3 = C
    det = (x2-x1)*(y3-y1) - (y2-y1)*(x3-x1)  # det([[B-A, C-A]])
    if det > eps: return 1   # CCW
    if det < -eps: return -1 # CW
    return 0                 # collinear

def triangle_area(A, B, C):
    x1,y1 = A; x2,y2 = B; x3,y3 = C
    return abs((x1*(y2-y3) + x2*(y3-y1) + x3*(y1-y2)))/2.0  # shoelace

A,B,C = (0.,0.), (3.,0.), (1.,2.)
print("orient2d:", orient2d(A,B,C))   # 1 (ë°˜ì‹œê³„)
print("â–³ABC area:", triangle_area(A,B,C))  # 3.0

def polygon_area(poly):
    """
    poly: [(x1,y1), (x2,y2), ..., (xn,yn)], n>=3, ë‹«íŒë‹¤ ê°€ì •(ë§ˆì§€ë§‰=ì²« ì  ì•„ë‹˜)

    s1 = sum(xs[i]*ys[i+1] for i in range(len(poly)))
    s2 = sum(ys[i]*xs[i+1] for i in range(len(poly)))
