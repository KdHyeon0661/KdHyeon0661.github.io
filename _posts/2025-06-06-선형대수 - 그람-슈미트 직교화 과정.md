import numpy as np

def gram_schmidt(V):
    U = []
    for v in V:
        proj = sum(np.dot(v, u) * u for u in U)
        u = v - proj
        u = u / np.linalg.norm(u)
        U.append(u)
    return np.array(U)

# ì˜ˆì‹œ
